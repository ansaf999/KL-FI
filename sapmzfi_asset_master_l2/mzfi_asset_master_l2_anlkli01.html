MODULE user_command_9001 INPUT.
  CASE sy-ucomm.
      ucomm = sy-ucomm.
    WHEN 'BACK' OR 'EXIT' OR 'CANCEL'.
      LEAVE PROGRAM.

    WHEN 'CREATE'.
*      DELETE IT_L2 WHERE ANLKL IS INITIAL.
      REFRESH it_l2.
      CLEAR wa_l2.
      CALL SCREEN 9000.

    WHEN 'DISPLAY' or 'CHANGE'.
      DATA : txk50 TYPE txt50_ankt,
             l1_desc TYPE zl1_desc.

      SELECT * FROM zfi_acm_l2 INTO CORRESPONDING FIELDS OF TABLE it_l2 WHERE anlkl IN s_anlkl
                                                                          AND l1_code IN sl1_code.
      IF sy-subrc NE 0.
        MESSAGE 'NO DATA EXIST' TYPE 'E'.
      ENDIF.

      LOOP AT it_l2 INTO wa_l2.
        SELECT SINGLE txk50 FROM ankt INTO txk50
                        WHERE anlkl = wa_l2-anlkl
                          AND spras = 'E'.
        wa_l2-txk50 = txk50.

        SELECT SINGLE l1_desc FROM zfi_acm_l1 INTO l1_desc
                        WHERE anlkl = wa_l2-anlkl
                          AND l1_code = wa_l2-l1_code.
        wa_l2-l1_desc = l1_desc.

        MODIFY it_l2 FROM wa_l2 TRANSPORTING txk50 l1_desc WHERE anlkl = wa_l2-anlkl
                                                            AND l1_code = wa_l2-l1_code..
        CLEAR : wa_l2, txk50, l1_desc.
      ENDLOOP.

      CALL SCREEN 9000.
  ENDCASE.
ENDMODULE. " USER_COMMAND_9001 INPUT
