FORM get_data .

  IF s_prctr IS NOT INITIAL.
    SELECT * FROM zfi_prj_prctr INTO TABLE it_prctr
      WHERE bukrs IN s_bukrs
       AND  prctr IN s_prctr
       AND  project IN s_prj.
    IF it_prctr IS NOT INITIAL.
      SELECT * FROM t001 INTO TABLE it_t001 FOR ALL ENTRIES IN
        it_prctr WHERE bukrs = it_prctr-bukrs.

      SELECT * FROM zfi_prjmapp INTO TABLE it_prjmapp FOR ALL ENTRIES IN
       it_prctr WHERE bukrs =  it_prctr-bukrs
                AND project =  it_prctr-project
               AND prstdt LE p_repdt
              AND prendt GE p_repdt.

      SELECT * FROM zfi_prjbnkzapp INTO TABLE it_prjbank FOR ALL ENTRIES IN
             it_prctr WHERE project =  it_prctr-project
                  AND prctr =  it_prctr-prctr.
      IF it_prjbank IS NOT INITIAL.
        SELECT * FROM zfi_signzapp INTO TABLE it_sign FOR ALL ENTRIES IN
          it_prjbank WHERE hbkid =   it_prjbank-hbkid.
      ENDIF.
    ENDIF.
  ELSE.
    SELECT * FROM zfi_prjmapp INTO TABLE it_prjmapp
      WHERE bukrs IN s_bukrs AND project IN s_prj
      AND prstdt LE p_repdt
       AND prendt GE p_repdt.

    IF it_prjmapp IS NOT INITIAL.

      SELECT * FROM zfi_prj_prctr INTO TABLE it_prctr FOR ALL ENTRIES IN
        it_prjmapp WHERE bukrs = it_prjmapp-bukrs
         AND  project = it_prjmapp-project.

      SELECT * FROM t001 INTO TABLE it_t001 FOR ALL ENTRIES IN
         it_prjmapp WHERE bukrs = it_prjmapp-bukrs.

      SELECT * FROM zfi_prjbnkzapp INTO TABLE it_prjbank FOR ALL ENTRIES IN
             it_prjmapp WHERE project =  it_prjmapp-project.
      IF it_prjbank IS NOT INITIAL.
        SELECT * FROM zfi_signzapp INTO TABLE it_sign FOR ALL ENTRIES IN
          it_prjbank WHERE hbkid =   it_prjbank-hbkid.
      ENDIF.
    ENDIF.
  ENDIF.
ENDFORM. " GET_DATA

FORM process_data .
  IF it_prjmapp IS NOT INITIAL.

                         WITH KEY bukrs = wa_final-bukrs.
      IF sy-subrc EQ 0.
        wa_final-butxt = wa_t001-butxt.
      ENDIF.
      READ TABLE it_prctr INTO wa_prctr
                          WITH KEY bukrs   = wa_final-bukrs
                                   project = wa_final-project.
      IF sy-subrc EQ 0.
        wa_final-prctr = wa_prctr-prctr.
        wa_final-ltext = wa_prctr-ltext.
      ENDIF.
      READ TABLE it_prjbank INTO wa_prjbank
                            WITH KEY project = wa_final-project
                                     prctr = wa_final-prctr.
      IF sy-subrc EQ 0.
        wa_final-hbkid = wa_prjbank-hbkid.
        SELECT SINGLE bankl
          FROM t012
          INTO wa_final-bankl WHERE
          bukrs = wa_final-bukrs AND
          hbkid = wa_final-hbkid AND
          banks = 'IN'.
        CLEAR wa_sign.
        READ TABLE it_sign INTO wa_sign WITH KEY bukrs = wa_final-bukrs
                                                 hbkid = wa_final-hbkid
                                                 sign_ty = '01'.
        IF sy-subrc EQ 0.
          CLEAR: wa_usr21.
          wa_final-sign1 = wa_sign-uname.

          SELECT SINGLE name_textc FROM user_addr INTO wa_final-sig1n
            WHERE bname = wa_final-sign1.

          SELECT SINGLE * FROM usr21 INTO wa_usr21
         WHERE bname = wa_final-sign1.
          IF sy-subrc EQ 0.
            SELECT SINGLE smtp_addr FROM adr6 INTO wa_final-sig1e
              WHERE addrnumber  = wa_usr21-addrnumber
                 AND persnumber = wa_usr21-persnumber.
          ENDIF.
          CLEAR: wa_usr21.
        ENDIF.
        CLEAR wa_sign.
        READ TABLE it_sign INTO wa_sign WITH KEY bukrs = wa_final-bukrs
                                                 hbkid = wa_final-hbkid
                                                 sign_ty = '02'.
        IF sy-subrc EQ 0.
          CLEAR: wa_usr21.
          wa_final-sign2 = wa_sign-uname.

          SELECT SINGLE name_textc FROM user_addr INTO wa_final-sig2n
            WHERE bname = wa_final-sign2.

          SELECT SINGLE * FROM usr21 INTO wa_usr21
         WHERE bname = wa_final-sign2.
          IF sy-subrc EQ 0.
            SELECT SINGLE smtp_addr FROM adr6 INTO wa_final-sig2e
              WHERE addrnumber  = wa_usr21-addrnumber
                 AND persnumber = wa_usr21-persnumber.
          ENDIF.

        ENDIF.
      ENDIF.
      APPEND wa_final TO it_final.
      CLEAR: wa_final, wa_usr21,  wa_prjbank, wa_sign, wa_prjmapp,
             wa_t001,wa_prctr.
    ENDLOOP.
  ELSEIF it_prctr IS NOT INITIAL.
    LOOP AT it_prctr INTO wa_prctr.

      wa_final-bukrs = wa_prctr-bukrs.
      wa_final-prctr = wa_prctr-project.
      wa_final-pr_desc = wa_prctr-pr_desc.
      READ TABLE it_t001 INTO wa_t001 WITH KEY
      bukrs = wa_final-bukrs.
      IF sy-subrc EQ 0.
        wa_final-butxt = wa_t001-butxt.
      ENDIF.
      READ TABLE it_prjmapp INTO wa_prjmapp WITH KEY
                                 bukrs = wa_final-bukrs
                                 project = wa_final-project.
      IF sy-subrc EQ 0.
        wa_final-lcode = wa_prjmapp-authkey.
        wa_final-ldesc = wa_prjmapp-auth_desc.
      ENDIF.
      READ TABLE it_prjbank INTO wa_prjbank WITH KEY
                                     project = wa_final-project
                                     prctr = wa_final-prctr.
      IF sy-subrc EQ 0.
        wa_final-hbkid = wa_prjbank-hbkid.
        SELECT SINGLE bankl FROM t012 INTO wa_final-bankl WHERE
          bukrs = wa_final-bukrs AND
          hbkid = wa_final-hbkid AND
          banks = 'IN'.
        CLEAR wa_sign.
        READ TABLE it_sign INTO wa_sign WITH KEY bukrs = wa_final-bukrs
                                                 hbkid = wa_final-hbkid
                                                 sign_ty = '01'.
        IF sy-subrc EQ 0.
          CLEAR: wa_usr21.
          wa_final-sign1 = wa_sign-uname.

          SELECT SINGLE name_textc FROM user_addr INTO wa_final-sig1n
            WHERE bname = wa_final-sign1.

          SELECT SINGLE * FROM usr21 INTO wa_usr21
         WHERE bname = wa_final-sign1.
          IF sy-subrc EQ 0.
            SELECT SINGLE smtp_addr FROM adr6 INTO wa_final-sig1e
              WHERE addrnumber  = wa_usr21-addrnumber
                AND persnumber = wa_usr21-persnumber.
          ENDIF.
          CLEAR: wa_usr21.
        ENDIF.
        CLEAR wa_sign.
        READ TABLE it_sign INTO wa_sign WITH KEY bukrs = wa_final-bukrs
                                                 hbkid = wa_final-hbkid

        IF sy-subrc EQ 0.

          IF sy-subrc EQ 0.

                 AND persnumber = wa_usr21-persnumber.
          ENDIF.
          CLEAR: wa_usr21.
        ENDIF.
      ENDIF.
      APPEND wa_final TO it_final.
      CLEAR: wa_final, wa_usr21,  wa_prjbank, wa_sign, wa_prjmapp,
             wa_t001,wa_prctr.
    ENDLOOP.
  ENDIF.
