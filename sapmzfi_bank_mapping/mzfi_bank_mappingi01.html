MODULE user_command_9000 INPUT.

  IF sy-ucomm EQ 'EDIT' OR
     sy-ucomm EQ 'DISPLAY'.

    SELECT * FROM zfi_bank_mapping INTO TABLE it_mapping
                       WHERE bukrs IN s_bukrs
                        AND  hbkid IN s_hbkid.

    LOOP AT it_mapping INTO wa_mapping.
      lv_tabix = sy-tabix.
      IF wa_mapping-bukrs IS NOT INITIAL.
        SELECT SINGLE * FROM t001 INTO wa_t001
               WHERE bukrs = wa_mapping-bukrs.

        SELECT SINGLE name1 FROM adrc INTO wa_mapping-butxt
                   WHERE addrnumber = wa_t001-adrnr.
      ENDIF.

      IF NOT wa_mapping-hbkid IS INITIAL.
        SELECT SINGLE text1 FROM t012t INTO wa_mapping-text1
               WHERE hbkid = wa_mapping-hbkid.
      ENDIF.

      SELECT SINGLE bankn FROM t012k INTO wa_mapping-bankn
             WHERE bukrs = wa_mapping-bukrs
               AND hbkid = wa_mapping-hbkid
               AND hktid = wa_mapping-hktid.

      IF NOT wa_mapping-code_catg IS INITIAL.
        SELECT SINGLE catg_desc FROM zfi_bank_accode INTO wa_mapping-catg_desc
               WHERE code_catg = wa_mapping-code_catg.
      ENDIF.

      MODIFY it_mapping FROM wa_mapping INDEX lv_tabix
                        TRANSPORTING butxt text1 bankn catg_desc.
    ENDLOOP.
    CALL SCREEN 9001.

  ELSEIF sy-ucomm EQ 'BACK' OR
         sy-ucomm EQ 'EXIT' OR
         sy-ucomm EQ 'CANC'.
    LEAVE PROGRAM.
  ENDIF.
ENDMODULE. " USER_COMMAND_9000 INPUT
