MODULE user_command_9000 INPUT.

  CASE sy-ucomm.

      flag = 'X'.

      SELECT * FROM zfi_banksig INTO TABLE it_banksig
                   WHERE bukrs IN s_bukrs
                     AND hbkid IN s_hbkid.

        lv_tabix = sy-tabix.
        IF wa_banksig-bukrs IS NOT INITIAL.
          SELECT SINGLE * FROM t001 INTO wa_t001
                 WHERE bukrs = wa_banksig-bukrs.

          SELECT SINGLE name1 FROM adrc INTO wa_banksig-butxt
                     WHERE addrnumber = wa_t001-adrnr.
        ENDIF.

        IF NOT wa_banksig-hbkid IS INITIAL.
          SELECT SINGLE text1 FROM t012t INTO wa_banksig-text1
                 WHERE hbkid = wa_banksig-hbkid.
        ENDIF.

        IF NOT wa_banksig-sign_code IS INITIAL.
          SELECT SINGLE sign_desc FROM zfi_signatories INTO wa_banksig-sign_desc
                 WHERE sign_code = wa_banksig-sign_code.
        ENDIF.

        SELECT SINGLE bankn FROM t012k INTO wa_banksig-bankn
                    WHERE bukrs = wa_banksig-bukrs
                      AND hbkid = wa_banksig-hbkid

        CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'

        MODIFY it_banksig FROM wa_banksig INDEX lv_tabix
                         TRANSPORTING butxt text1 sign_desc bankn name1.

MODULE user_command_9001 INPUT.

  CASE sy-ucomm.
    WHEN 'SAVE'.
      IF flag = 'X'.
        LOOP AT it_banksig INTO wa_banksig.

          IF wa_banksig-bukrs IS INITIAL OR
             wa_banksig-hbkid IS INITIAL OR
             wa_banksig-hktid IS INITIAL OR
             wa_banksig-sign_code IS INITIAL OR
             wa_banksig-vbund IS INITIAL OR
             wa_banksig-start_date IS INITIAL OR
             wa_banksig-limit IS INITIAL .
            MESSAGE 'Pls fill in the required fields' TYPE 'E'.
          ENDIF.

          SELECT SINGLE * FROM zfi_banksig INTO wa_zfi_banksig
            WHERE bukrs = wa_banksig-bukrs
              AND hbkid = wa_banksig-hbkid
              AND hktid = wa_banksig-hktid
              AND sign_code = wa_banksig-sign_code
              AND vbund = wa_banksig-vbund
              AND start_date = wa_banksig-start_date.

          IF wa_zfi_banksig IS NOT INITIAL.
            MESSAGE 'Record already exist' TYPE 'E'.

          ELSEIF wa_zfi_banksig IS INITIAL.
            INSERT zfi_banksig FROM wa_banksig.
            MESSAGE 'Data saved successfully' TYPE 'S'.
          ENDIF.
        ENDLOOP.

      ELSEIF flag = ' '.
        LOOP AT it_banksig INTO wa_banksig.
          SELECT SINGLE * FROM zfi_banksig INTO wa_zfi_banksig
            WHERE bukrs = wa_banksig-bukrs
              AND hbkid = wa_banksig-hbkid
              AND hktid = wa_banksig-hktid
              AND sign_code = wa_banksig-sign_code
              AND vbund = wa_banksig-vbund
              AND start_date = wa_banksig-start_date.

          IF wa_zfi_banksig IS NOT INITIAL.
            UPDATE zfi_banksig FROM wa_banksig.
            MESSAGE 'Data updated successfully' TYPE 'S'.
          ENDIF.
        ENDLOOP.
      ENDIF.

    WHEN 'BACK' OR 'EXIT' OR 'CANC'.

  ENDCASE.
