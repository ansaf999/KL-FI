FORM checkgrid .

  CLEAR : index,
          pcode,
          cumle,
          conte,
          zperc,
          conop.

  LOOP AT it_pmast INTO wa_pmast.

    LOOP AT it_pmast INTO wa_pmast2 FROM index + 1.

      IF  wa_pmast-zperc = wa_pmast2-zperc
      and wa_pmast-cumle = wa_pmast2-cumle
      AND wa_pmast-conte = wa_pmast2-conte
      AND wa_pmast-conop = wa_pmast2-conop.

        pcode    = wa_pmast-pcode.

        conte    = wa_pmast-conte.
        conop    = wa_pmast-conop.

        CONCATENATE ' You entered same combination for prefcode ' pcode  ' and ' wa_pmast2-pcode
        INTO str SEPARATED BY ''.
        MESSAGE str TYPE 'E'.
      endif.
      ENDIF.

      CLEAR:wa_pmast2,
            pcode,
            cumle,
            conte,
            conop.
    ENDLOOP.
