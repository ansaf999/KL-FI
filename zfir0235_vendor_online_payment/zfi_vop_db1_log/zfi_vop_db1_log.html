FUNCTION ZFI_VOP_DB1_LOG.
*"--------------------------------------------------------------------

* Global data declarations

  FIELD-SYMBOLS: <fs_log> TYPE zfi_zpp_postl.
  IF it_tlog[] IS NOT INITIAL.
    LOOP AT it_tlog ASSIGNING <fs_log>.
      CALL FUNCTION 'NUMBER_GET_NEXT'
        EXPORTING
          nr_range_nr = '01'
          object      = 'ZFI_TLOG'
        IMPORTING
          number      = <fs_log>-chlno.

      <fs_log>-chngdt = sy-datum.
      <fs_log>-chngtm = sy-uzeit.
      <fs_log>-usnam  = sy-uname.
      <fs_log>-mandt = sy-mandt.
    ENDLOOP.

    INSERT zfi_zpp_postl FROM TABLE it_tlog.
    WAIT UP TO 2 SECONDS.
    REFRESH it_tlog.
  ENDIF.



ENDFUNCTION.
